#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#define N 10
int main()
{
	int i;
	int x;
	printf("input the x:\n");
	scanf("%d",&x);
	int a[N];
	int n;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	int result=a[n-1];
	for(i=n-1;i>0;i--)
	{
		result=a[i-1]+x*result;
	}
	printf("%d",result);
	return 0;
}
